1//! Deskulpt logs manager and its APIs.
2
3use std::path::{Path, PathBuf};
4
5use anyhow::Result;
6use tauri::{AppHandle, Manager, Runtime};
7use tracing::Level;
8use tracing_appender::non_blocking::{NonBlockingBuilder, WorkerGuard};
9use tracing_appender::rolling::{RollingFileAppender, Rotation};
10use tracing_subscriber::filter::Targets;
11use tracing_subscriber::fmt::time::UtcTime;
12use tracing_subscriber::layer::SubscriberExt;
13use tracing_subscriber::{Layer, Registry, fmt};
14
15use crate::reader::{Cursor, Page, RollingTailReader};
16
17/// Manager for Deskulpt logs.
18pub struct LogsManager<R: Runtime> {
19    /// The Tauri app handle.
20    _app_handle: AppHandle<R>,
21    /// The directory where log files are stored.
22    dir: PathBuf,
23    /// A guard that flushes pending logs when dropped.
24    _guard: WorkerGuard,
25}
26
27impl<R: Runtime> LogsManager<R> {
28    /// Initialize the logging system.
29    ///
30    /// This will set up structured logging in newline-delimited JSON format
31    /// with daily rotation, retaining up to 10 log files. The logging system
32    /// remains active for the lifetime of the manager.
33    pub fn new(app_handle: AppHandle<R>) -> Result<Self> {
34        let dir = app_handle.path().app_log_dir()?;
35        std::fs::create_dir_all(&dir)?;
36
37        let appender = RollingFileAppender::builder()
38            .rotation(Rotation::DAILY)
39            .max_log_files(10)
40            .filename_prefix("deskulpt")
41            .filename_suffix("log")
42            .build(&dir)?;
43
44        let (writer, guard) = NonBlockingBuilder::default().finish(appender);
45
46        let file_layer = fmt::layer()
47            .json()
48            .with_target(true)
49            .with_file(true)
50            .with_line_number(true)
51            .with_timer(UtcTime::rfc_3339())
52            .with_current_span(false)
53            .with_span_list(true)
54            .flatten_event(true)
55            .with_writer(writer)
56            .with_filter(
57                Targets::new()
58                    .with_target("deskulpt", Level::TRACE)
59                    .with_target("frontend::canvas", Level::TRACE)
60                    .with_target("frontend::manager", Level::TRACE),
61            );
62
63        let subscriber = Registry::default().with(file_layer);
64        tracing::subscriber::set_global_default(subscriber)?;
65
66        // Set up panic hook to log uncaught panics
67        let previous_hook = std::panic::take_hook();
68        std::panic::set_hook(Box::new(move |panic_info| {
69            tracing_panic::panic_hook(panic_info);
70            previous_hook(panic_info);
71        }));
72
73        Ok(Self {
74            dir,
75            _app_handle: app_handle,
76            _guard: guard,
77        })
78    }
79
80    /// Get the directory where log files are stored.
81    pub fn dir(&self) -> &Path {
82        &self.dir
83    }
84
85    /// Collect log files in most recent first order.
86    fn collect(&self) -> Result<Vec<PathBuf>> {
87        let mut files = std::fs::read_dir(&self.dir)?
88            .filter_map(|entry| {
89                let entry = entry.ok()?;
90                let path = entry.path();
91                let name = path.file_name()?.to_string_lossy();
92                if name.starts_with("deskulpt.") && name.ends_with(".log") {
93                    Some(path)
94                } else {
95                    None
96                }
97            })
98            .collect::<Vec<_>>();
99
100        // Here we assume that the filenames are timestamps, so sorting by
101        // filename in descending order should correspond to most recent first
102        files.sort_by(|a, b| b.file_name().cmp(&a.file_name()));
103        Ok(files)
104    }
105
106    /// Read a page of log entries.
107    ///
108    /// This will read up to `limit` log entries with severity at or above
109    /// `min_level`. If `cursor` is `None`, this method starts reading from the
110    /// newest entries. Otherwise, it continues reading from the provided
111    /// cursor, which should have been obtained from a previous call to this
112    /// method.
113    pub fn read(&self, limit: usize, min_level: Level, cursor: Option<Cursor>) -> Result<Page> {
114        let files = self.collect()?;
115        let mut reader = RollingTailReader::new(files, min_level);
116        reader.read(limit, cursor)
117    }
118
119    /// Clear all log files.
120    ///
121    /// The latest log file is truncated instead of deleted to ensure that
122    /// logging can continue without interruption. All older log files are
123    /// permanently deleted. The total amount of space freed is returned in
124    /// bytes.
125    ///
126    /// This method returns an error if log file collection fails in the first
127    /// place. Individual file deletion or truncation failures are silently
128    /// ignored, and they do not contribute to the computed freed space.
129    pub fn clear(&self) -> Result<u64> {
130        let log_files = self.collect()?;
131
132        let mut freed_space: u64 = log_files
133            .iter()
134            .skip(1)
135            .filter_map(|file| {
136                let size = file.metadata().ok().map(|m| m.len());
137                std::fs::remove_file(file).ok().and(size)
138            })
139            .sum();
140
141        if let Some(latest_file) = log_files.first() {
142            let size = latest_file.metadata().map_or(0, |m| m.len());
143            if std::fs::OpenOptions::new()
144                .write(true)
145                .truncate(true)
146                .open(latest_file)
147                .is_ok()
148            {
149                freed_space += size;
150            }
151        }
152
153        Ok(freed_space)
154    }
155}
