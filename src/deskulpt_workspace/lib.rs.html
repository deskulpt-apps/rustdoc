7use std::path::PathBuf;
8
9/// Get the root directory of the workspace.
10pub fn root_dir() -> PathBuf {
11    PathBuf::from(env!("WORKSPACE_DIR"))
12}
13
14/// Get the docs directory of the workspace.
15pub fn docs_dir() -> PathBuf {
16    root_dir().join("docs")
17}
18
19/// Get the directory of a crate in the workspace.
20pub fn crate_dir(crate_name: &str) -> PathBuf {
21    root_dir().join("crates").join(crate_name)
22}
23
24/// Get the directory of a package in the workspace.
25pub fn package_dir(package_name: &str) -> PathBuf {
26    root_dir().join("packages").join(package_name)
27}
