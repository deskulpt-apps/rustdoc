1use serde::Serialize;
2
3/// Serializable wrapper around [`anyhow::Error`].
4///
5/// This implements [`Serialize`] with the [`Debug`] representation of the
6/// error. Any error that can be converted into an [`anyhow::Error`] can be
7/// converted into this error type, meaning that error propagation with `?`
8/// works in the same way as with [`anyhow::Error`].
9#[derive(Debug)]
10pub struct SerError(anyhow::Error);
11
12impl<E> From<E> for SerError
13where
14    E: Into<anyhow::Error>,
15{
16    fn from(err: E) -> Self {
17        SerError(err.into())
18    }
19}
20
21impl Serialize for SerError {
22    fn serialize<S>(&self, serializer: S) -> std::result::Result<S::Ok, S::Error>
23    where
24        S: serde::ser::Serializer,
25    {
26        serializer.serialize_str(format!("{:?}", self.0).as_str())
27    }
28}
29
30impl specta::Type for SerError {
31    fn inline(
32        type_map: &mut specta::TypeCollection,
33        generics: specta::Generics,
34    ) -> specta::datatype::DataType {
35        <String as specta::Type>::inline(type_map, generics)
36    }
37
38    fn reference(
39        type_map: &mut specta::TypeCollection,
40        generics: &[specta::datatype::DataType],
41    ) -> specta::datatype::reference::Reference {
42        <String as specta::Type>::reference(type_map, generics)
43    }
44}
45
46/// Result type with [`SerError`].
47///
48/// This is serializable as long as `T` is serializable.
49pub type SerResult<T> = Result<T, SerError>;

51#[doc(hidden)]
52#[macro_export]
53macro_rules! __ser_bail {
54    ($msg:literal $(,)?) => {
55        return Err(anyhow::anyhow!($msg).into())
56    };
57    ($err:expr $(,)?) => {{
58        return Err(anyhow::anyhow!($err).into())
59    }};
60    ($fmt:expr, $($arg:tt)*) => {
61        return Err(anyhow::anyhow!($fmt, $($arg)*).into())
62    };
63}
