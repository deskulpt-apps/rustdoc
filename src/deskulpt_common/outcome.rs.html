1use serde::{Deserialize, Serialize};

3/// A result-like binary outcome.
4///
5/// This represents the outcome of an operation that can either succeed with a
6/// value of type `T` or fail with an error message.
7#[derive(Debug, Clone, Serialize, Deserialize, specta::Type)]
8#[serde(tag = "type", content = "content", rename_all = "camelCase")]
9pub enum Outcome<T> {
10    Ok(T),
11    Err(String),
12}
13
14impl<T, E: std::fmt::Debug> From<Result<T, E>> for Outcome<T> {
15    fn from(result: Result<T, E>) -> Self {
16        match result {
17            Ok(value) => Outcome::Ok(value),
18            Err(e) => Outcome::Err(format!("{e:?}")),
19        }
20    }
21}
