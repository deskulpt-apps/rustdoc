1//! Deskulpt path utilities.
2
3use std::fs::create_dir_all;
4use std::path::{Path, PathBuf};
5use std::sync::Arc;
6
7use anyhow::{Result, anyhow};
8use once_cell::sync::OnceCell;
9use tauri::{App, AppHandle, Manager, Runtime};
10
11/// Thread-safe lazily-initialized static for the widgets directory.
12static WIDGETS_DIR: OnceCell<Arc<PathBuf>> = OnceCell::new();
13
14/// Thread-safe lazily-initialized static for the persistence directory.
15static PERSIST_DIR: OnceCell<Arc<PathBuf>> = OnceCell::new();
16
17/// Extension trait for path-related operations.
18pub trait PathExt<R: Runtime>: Manager<R> {
19    /// Initialize the widgets directory.
20    ///
21    /// This will create the widgets directory if it does not exist. It must be
22    /// called before calling the [`widgets_dir`](PathExt::widgets_dir) method.
23    fn init_widgets_dir(&self) -> Result<()> {
24        let widgets_dir = WIDGETS_DIR.get_or_init(|| {
25            let widgets_base_dir = if cfg!(debug_assertions) {
26                self.path().resource_dir().unwrap()
27            } else {
28                self.path().document_dir().unwrap().join("Deskulpt")
29            };
30            let widgets_base_dir = dunce::simplified(&widgets_base_dir);
31            Arc::new(widgets_base_dir.join("widgets"))
32        });
33
34        if !widgets_dir.exists() {
35            create_dir_all(widgets_dir.as_ref())?;
36        }
37        Ok(())
38    }
39
40    /// Get a reference to the widgets directory.
41    ///
42    /// This will create the widgets directory if it does not exist, which can
43    /// happen if one removes that directory during the application. This will
44    /// error if the [`init_widgets_dir`](PathExt::init_widgets_dir) method has
45    /// not been called.
46    fn widgets_dir(&self) -> Result<&Path> {
47        let widgets_dir = WIDGETS_DIR
48            .get()
49            .ok_or_else(|| anyhow!("`init_widgets_dir` must be called first"))?
50            .as_path();
51
52        if !widgets_dir.exists() {
53            create_dir_all(widgets_dir)?;
54        }
55        Ok(widgets_dir)
56    }
57
58    /// Get a reference to a widget directory.
59    fn widget_dir(&self, id: &str) -> Result<PathBuf> {
60        Ok(self.widgets_dir()?.join(id))
61    }
62
63    /// Initialize the persistence directory.
64    ///
65    /// This will create the persistence directory if it does not exist. It must
66    /// be called before calling the [`persist_dir`](PathExt::persist_dir)
67    /// method.
68    fn init_persist_dir(&self) -> Result<()> {
69        let persist_dir = PERSIST_DIR.get_or_init(|| {
70            let persist_dir = self.path().app_local_data_dir().unwrap();
71            Arc::new(persist_dir)
72        });
73
74        if !persist_dir.exists() {
75            create_dir_all(persist_dir.as_ref())?;
76        }
77        Ok(())
78    }
79
80    /// Get a reference to the persistence directory.
81    ///
82    /// This will create the persistence directory if it does not exist, which
83    /// can happen if one removes that directory during the application. This
84    /// will error if the [`init_persist_dir`](PathExt::init_persist_dir) method
85    /// has not been called.
86    fn persist_dir(&self) -> Result<&Path> {
87        let persist_dir = PERSIST_DIR
88            .get()
89            .ok_or_else(|| anyhow!("`init_persist_dir` must be called first"))?
90            .as_path();
91
92        if !persist_dir.exists() {
93            create_dir_all(persist_dir)?;
94        }
95        Ok(persist_dir)
96    }
97}
98
99impl<R: Runtime> PathExt<R> for App<R> {}
100impl<R: Runtime> PathExt<R> for AppHandle<R> {}
