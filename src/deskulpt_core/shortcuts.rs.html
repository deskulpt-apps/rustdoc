12/// Re-register a shortcut.
13///
14/// The old shortcut will be unregistered and the new shortcut will be
15/// registered, with the listener determined by the shortcut action.
16fn reregister_shortcut<R: Runtime>(
17    gs: &GlobalShortcut<R>,
18    action: &ShortcutAction,
19    old: Option<&String>,
20    new: Option<&String>,
21) -> Result<()> {
22    if let Some(shortcut) = old {
23        gs.unregister(shortcut.as_str())?;
24    }
25
26    let handler: fn(&AppHandle<R>) = match action {
27        ShortcutAction::ToggleCanvasImode => |app_handle| {
28            if let Err(e) = app_handle.toggle_canvas_imode() {
29                error!("Failed to toggle canvas interaction mode: {e}");
30            }
31        },
32        ShortcutAction::OpenManager => |app_handle| {
33            if let Err(e) = app_handle.open_manager() {
34                error!("Failed to open the manager window: {e}");
35            }
36        },
37    };
38
39    if let Some(shortcut) = new {
40        gs.on_shortcut(shortcut.as_str(), move |app_handle, _, event| {
41            if event.state == ShortcutState::Pressed {
42                handler(app_handle);
43            }
44        })?;
45    }
46
47    Ok(())
48}

50/// Extension trait for keyboard shortcut operations.
51pub trait ShortcutsExt<R: Runtime>: Manager<R> + SettingsExt<R> + GlobalShortcutExt<R> {
52    /// Initialize keyboard shortcuts management.
53    ///
54    /// This immediately registers shortcuts based on the settings. Failure to
55    /// register the shortcuts are properly logged but not fatal. It also
56    /// re-registers shortcuts when shortcuts in the settings change.
57    fn init_shortcuts(&self) {
58        {
59            let gs = self.global_shortcut();
60            let settings = self.settings().read();
61            for (action, shortcut) in &settings.shortcuts {
62                if let Err(e) = reregister_shortcut(gs, action, None, Some(shortcut)) {
63                    error!("Failed to register shortcut {shortcut:?} for {action:?}: {e:?}");
64                }
65            }
66        }
67
68        let app_handle = self.app_handle().clone();
69        self.settings().on_shortcut_change(move |action, old, new| {
70            let gs = app_handle.global_shortcut();
71            if let Err(e) = reregister_shortcut(gs, action, old, new) {
72                error!(
73                    "Failed to re-register shortcut from {old:?} to {new:?} for {action:?}: {e:?}"
74                );
75            }
76        });
77    }
78}
