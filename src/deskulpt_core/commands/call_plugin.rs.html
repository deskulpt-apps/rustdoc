19/// The Deskulpt core should keep a state of the registered plugins and call the
20/// plugins dynamically. Also, instead of invoking the plugins directly, the
21/// Deskulpt core should not depend on any of the plugins and should use IPC to
22/// communicate with the plugins.
23///
24/// Also, in order to simplify the engine API for the plugin (because it is
25/// a temporary implementation), `app_handle` is using the default runtime but
26/// it should be a generic `R: Runtime` parameter in the final implementation.
27#[command]
28#[specta::specta]
29pub async fn call_plugin<R: Runtime>(
30    app_handle: AppHandle<R>,
31    plugin: String,
32    command: String,
33    id: String,
34    payload: Option<serde_json::Value>,
35) -> SerResult<serde_json::Value> {
36    let widget_dir_fn = move |id: &str| app_handle.widgets().dir().join(id);
37
38    match plugin.as_str() {
39        "fs" => {
40            let plugin = FS_PLUGIN.lock();
41            let result = deskulpt_plugin::call_plugin(
42                widget_dir_fn,
43                &*plugin,
44                command.as_str(),
45                id,
46                payload,
47            )?;
48            Ok(result)
49        },
50        "sys" => {
51            let plugin = SYS_PLUGIN.lock();
52            let result = deskulpt_plugin::call_plugin(
53                widget_dir_fn,
54                &*plugin,
55                command.as_str(),
56                id,
57                payload,
58            )?;
59            Ok(result)
60        },
61        _ => ser_bail!("Unknown plugin: {}", plugin),
62    }
63}
