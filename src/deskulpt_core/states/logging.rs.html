17/// Maximum number of log files to retain.
18const MAX_LOG_FILES: usize = 10;
19
20/// Managed state for logging.
21struct LoggingState {
22    /// Guard that flushes the tracing worker on drop.
23    _guard: WorkerGuard,
24}
25
26/// Extension trait for operations related to logging.
27pub trait LoggingStateExt<R: Runtime>: Manager<R> + PathExt<R> {
28    /// Initialize state management for logging.
29    ///
30    /// This will set up structured logging in newline-delimited JSON format
31    /// with daily rotation and maximum [`MAX_LOG_FILES`] log files retained.
32    /// A panic hook is also set up to log uncaught panics.
33    fn manage_logging(&self) -> Result<()> {
34        let logs_dir = self.logs_dir()?;
35
36        let appender = RollingFileAppender::builder()
37            .rotation(Rotation::DAILY)
38            .max_log_files(MAX_LOG_FILES)
39            .filename_prefix("deskulpt")
40            .filename_suffix("log")
41            .build(logs_dir)?;
42        let (writer, guard) = NonBlockingBuilder::default().finish(appender);
43
44        let file_layer = fmt::layer()
45            .json()
46            .with_target(true)
47            .with_file(true)
48            .with_line_number(true)
49            .with_timer(UtcTime::rfc_3339())
50            .with_current_span(false)
51            .with_span_list(true)
52            .flatten_event(true)
53            .with_writer(writer.clone())
54            .with_filter(
55                Targets::new()
56                    .with_target("deskulpt", Level::TRACE)
57                    .with_target("frontend::canvas", Level::TRACE)
58                    .with_target("frontend::manager", Level::TRACE),
59            );
60
61        let subscriber = Registry::default().with(file_layer);
62        tracing::subscriber::set_global_default(subscriber)?;
63
64        let previous_hook = std::panic::take_hook();
65        std::panic::set_hook(Box::new(move |panic_info| {
66            tracing_panic::panic_hook(panic_info);
67            previous_hook(panic_info);
68        }));
69
70        self.manage(LoggingState { _guard: guard });
71        Ok(())
72    }
73
74    /// Discover log files and return their paths by newest first.
75    ///
76    /// This looks for log files in the logs directory with names matching the
77    /// pattern `deskulpt.*.log`, where `*` should be a timestamp though this is
78    /// not verified here. The returned list is sorted by filename in descending
79    /// order, which should correspond to most recent first if the `*`s are
80    /// indeed timestamps.
81    fn collect_logs(&self) -> Result<Vec<PathBuf>> {
82        let logs_dir = self.logs_dir()?;
83
84        let mut files = std::fs::read_dir(logs_dir)?
85            .filter_map(|entry| {
86                let entry = entry.ok()?;
87                let path = entry.path();
88                let name = path.file_name()?.to_string_lossy();
89                if name.starts_with("deskulpt.") && name.ends_with(".log") {
90                    Some(path)
91                } else {
92                    None
93                }
94            })
95            .collect::<Vec<_>>();
96
97        files.sort_by(|a, b| b.file_name().cmp(&a.file_name()));
98        Ok(files)
99    }
100
101    /// Clear all log files and return the freed disk space in bytes.
102    ///
103    /// The latest log file is truncated instead of deleted to ensure that
104    /// logging can continue without interruption. All older log files are
105    /// deleted. The total freed disk space is returned.
106    ///
107    /// Note that failure to delete or truncate a log file will not result in an
108    /// error, but will not contribute to the computed freed space. Failure to
109    /// discover the log files in the first place (before actual clearing
110    /// begins), however, willl result in an error.
111    fn clear_logs(&self) -> Result<u64> {
112        let log_files = self.collect_logs()?;
113
114        let mut total_size: u64 = log_files
115            .iter()
116            .skip(1)
117            .filter_map(|file| {
118                let size = file.metadata().ok().map(|m| m.len());
119                std::fs::remove_file(file).ok().and(size)
120            })
121            .sum();
122
123        if let Some(latest_file) = log_files.first() {
124            let size = latest_file.metadata().map(|m| m.len()).unwrap_or(0);
125            if std::fs::OpenOptions::new()
126                .write(true)
127                .truncate(true)
128                .open(latest_file)
129                .is_ok()
130            {
131                total_size += size;
132            }
133        }
134
135        Ok(total_size)
136    }
137}
