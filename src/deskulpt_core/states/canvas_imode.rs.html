17/// Layout information of the canvas window.
18#[derive(Copy, Clone)]
19struct CanvasLayout {
20    /// Physical x-coordinate.
21    x: f64,
22    /// Physical y-coordinate.
23    y: f64,
24    /// Inverse of the scale factor.
25    inv_scale: f64,
26}
27
28/// Managed state for canvas interaction mode.
29struct CanvasImodeState {
30    /// Lock for serializing `set_ignore_cursor_events` calls.
31    lock: RwLock<()>,
32    /// Layout information of the canvas window.
33    ///
34    /// We use [`SeqLock`] here for low-overhead and lock-free reads in the
35    /// global mousemove event listener which cannot afford blocking, thanks to
36    /// the fact that [`CanvasLayout`] is [`Copy`]. Writers must be rare, which
37    /// is the case here since they only happen when the canvas window is moved
38    /// or rescaled, mostly on startup.
39    layout: SeqLock<CanvasLayout>,
40}
41
42/// Whether the global mousemove listener is enabled.
43static LISTENING_MOUSEMOVE: AtomicBool = AtomicBool::new(false);
44
45/// Extension trait for operations on canvas interaction mode.
46pub trait CanvasImodeStateExt<R: Runtime>: Manager<R> + SettingsExt<R> {
47    /// Initialize state management for canvas interaction mode.
48    ///
49    /// This will also hook into settings changes and global mousemove events
50    /// and update the canvas window's interaction mode accordingly.
51    fn manage_canvas_imode(&self) -> Result<()> {
52        let canvas = DeskulptWindow::Canvas.webview_window(self)?;
53        let canvas_position = canvas.inner_position()?;
54        let canvas_layout = CanvasLayout {
55            x: canvas_position.x as f64,
56            y: canvas_position.y as f64,
57            inv_scale: 1.0 / canvas.scale_factor()?,
58        };
59        self.manage(CanvasImodeState {
60            lock: RwLock::new(()),
61            layout: SeqLock::new(canvas_layout),
62        });
63
64        let canvas_cloned = canvas.clone();
65        std::thread::spawn(move || {
66            // Delay the start of mousemove listener to avoid interfering with
67            // canvas window initialization, which is in most cases the heaviest
68            // period of writes to states that the mousemove listener may read;
69            // users commonly won't notice such delay because window creation
70            // and widgets rendering also take time.
71            std::thread::sleep(Duration::from_secs(1));
72
73            if let Err(e) = listen_to_mousemove(canvas_cloned) {
74                eprintln!("Failed to listen to global mousemove events: {}", e);
75            }
76        });
77
78        if self.settings().read().canvas_imode == CanvasImode::Auto {
79            LISTENING_MOUSEMOVE.store(true, Ordering::Release);
80        }
81
82        self.settings().on_canvas_imode_change(move |_, new| {
83            if let Err(e) = on_new_canvas_imode(&canvas, new) {
84                error!("Failed to update canvas interaction mode: {}", e);
85            }
86        });
87
88        Ok(())
89    }
90
91    /// Set the position of the canvas window.
92    ///
93    /// This should be called whenever the canvas window is moved.
94    fn set_canvas_position(&self, position: &PhysicalPosition<i32>) {
95        let state = self.state::<CanvasImodeState>();
96        let mut layout = state.layout.lock_write();
97        layout.x = position.x as f64;
98        layout.y = position.y as f64;
99    }
100
101    /// Set the scale factor of the canvas window.
102    ///
103    /// This should be called whenever the canvas window's scale factor changes.
104    fn set_canvas_scale_factor(&self, scale_factor: f64) {
105        let state = self.state::<CanvasImodeState>();
106        let mut layout = state.layout.lock_write();
107        layout.inv_scale = 1.0 / scale_factor;
108    }
109
110    /// Toggle the interaction mode of the canvas window.
111    ///
112    /// If the current mode is float or sink, it switches to the other mode. If
113    /// the current mode is auto, it is no-op since auto mode is not toggleable.
114    fn toggle_canvas_imode(&self) -> Result<()> {
115        self.settings().update_with(|settings| SettingsPatch {
116            canvas_imode: match settings.canvas_imode {
117                CanvasImode::Auto => None,
118                CanvasImode::Float => Some(CanvasImode::Sink),
119                CanvasImode::Sink => Some(CanvasImode::Float),
120            },
121            ..Default::default()
122        })?;
123        Ok(())
124    }
125}

128impl<R: Runtime> CanvasImodeStateExt<R> for AppHandle<R> {}

130/// Handler for canvas interaction mode changes.
131///
132/// This updates the canvas window's click-through state and the mousemove event
133/// listener's behavior according to the given mode. It also emits a toast
134/// notification to the canvas window, but failure to do so is non-fatal and
135/// will not result in an error.
136fn on_new_canvas_imode<R: Runtime>(canvas: &WebviewWindow<R>, mode: &CanvasImode) -> Result<()> {
137    match mode {
138        CanvasImode::Auto => {
139            LISTENING_MOUSEMOVE.store(true, Ordering::Release);
140        },
141        CanvasImode::Sink | CanvasImode::Float => {
142            // Set the flag with write lock acquired to avoid racing with the
143            // mousemove hook on setting `ignore_cursor_events`
144            let state = canvas.state::<CanvasImodeState>();
145            let _guard = state.lock.write();
146            LISTENING_MOUSEMOVE.store(false, Ordering::Release);
147            canvas.set_ignore_cursor_events(*mode == CanvasImode::Sink)?;
148        },
149    }
150
151    if let Err(e) = ShowToastEvent::Success(format!("Canvas interaction mode: {mode:?}"))
152        .emit_to(canvas, DeskulptWindow::Canvas)
153    {
154        error!("Failed to emit ShowToastEvent to canvas: {}", e);
155    }
156
157    Ok(())
158}

160/// Global mousemove event listener.
161///
162/// If the cheap check on [`LISTENING_MOUSEMOVE`] gives false, the hook will
163/// short-circuit immediately, effectively disabling the listener. Otherwise,
164/// it will check whether the mouse is over any widget in the canvas window. If
165/// so, the canvas will accept cursor events; otherwise, it will ignore them.
166fn listen_to_mousemove<R: Runtime>(canvas: WebviewWindow<R>) -> Result<()> {
167    let mut is_cursor_ignored = true;
168
169    global_mousemove::listen(move |event| {
170        if !LISTENING_MOUSEMOVE.load(Ordering::Acquire) {
171            return;
172        }
173
174        let state = canvas.state::<CanvasImodeState>();
175        let canvas_layout = state.layout.read();
176
177        let global_mousemove::MouseMoveEvent { x, y } = event;
178
179        // For macOS, mousemove coordinates are in logical coordinates, so
180        // only canvas physical position needs to be scaled
181        #[cfg(target_os = "macos")]
182        let scaled_x = x - canvas_layout.x * canvas_layout.inv_scale;
183        #[cfg(target_os = "macos")]
184        let scaled_y = y - canvas_layout.y * canvas_layout.inv_scale;
185
186        // For other platforms, mousemove coordinates are in physical
187        // coordinates, so they need to be scaled together with canvas position
188        #[cfg(not(target_os = "macos"))]
189        let scaled_x = (x - canvas_layout.x) * canvas_layout.inv_scale;
190        #[cfg(not(target_os = "macos"))]
191        let scaled_y = (y - canvas_layout.y) * canvas_layout.inv_scale;
192
193        let settings = match canvas.settings().try_read() {
194            Some(settings) => settings,
195            None => return, // Avoid blocking
196        };
197        let mouse_over_widget = settings.widgets.values().any(|widget| {
198            scaled_x >= widget.x as f64
199                && scaled_x < widget.x as f64 + widget.width as f64
200                && scaled_y >= widget.y as f64
201                && scaled_y < widget.y as f64 + widget.height as f64
202        });
203
204        // Avoid redundant calls by checking if the state has really changed
205        let should_ignore_cursor = !mouse_over_widget;
206        if should_ignore_cursor != is_cursor_ignored {
207            // Check the flag with read lock acquired to avoid racing with the
208            // writers on setting `ignore_cursor_events`
209            let state = canvas.state::<CanvasImodeState>();
210            let _guard = match state.lock.try_read() {
211                Some(guard) => guard,
212                None => return, // Avoid blocking
213            };
214
215            if !LISTENING_MOUSEMOVE.load(Ordering::Acquire) {
216                return;
217            }
218            is_cursor_ignored = should_ignore_cursor;
219            if let Err(e) = canvas.set_ignore_cursor_events(should_ignore_cursor) {
220                eprintln!("Failed to set cursor events state: {e}");
221            }
222        }
223    })?;
224
225    Ok(())
