21/// Initialize the internal Deskulpt widgets plugin.
22pub fn init<R: Runtime>() -> TauriPlugin<R> {
23    deskulpt_common::init::init_builder!()
24        .setup(|app_handle, _| {
25            app_handle.manage(WidgetsManager::new(app_handle.clone()));
26            Ok(())
27        })
28        .build()
29}
30
31/// Extension to [`Manager`] for accessing Deskulpt widgets APIs.
32pub trait WidgetsExt<R: Runtime> {
33    /// Get a reference to the [`WidgetsManager`] to access the APIs.
34    fn widgets(&self) -> &WidgetsManager<R>;
35}
36
37impl<R: Runtime, M: Manager<R>> WidgetsExt<R> for M {
38    fn widgets(&self) -> &WidgetsManager<R> {
39        self.state::<WidgetsManager<R>>().inner()
40    }
41}
