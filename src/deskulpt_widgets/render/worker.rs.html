3use anyhow::Result;
4use deskulpt_common::event::Event;
5use deskulpt_common::window::DeskulptWindow;
6use tauri::{AppHandle, Runtime};
7use tokio::sync::mpsc;
8use tracing::error;
9
10use crate::WidgetsExt;
11use crate::events::RenderEvent;
12use crate::render::bundler::Bundler;
13
14/// Tasks that the render worker can process.
15#[derive(Debug)]
16pub enum RenderWorkerTask {
17    /// Bundle and render a widget.
18    ///
19    /// The worker will use [`Bundler`] to bundle the specified widget at the
20    /// specified entry file. Upon completion, a [`RenderEvent`] will be emitted
21    /// to the canvas with the bundling result, whether success or
22    /// failure.
23    Render {
24        /// The widget ID.
25        id: String,
26        /// The entry file path relative to the root of the widget.
27        entry: String,
28    },
29}
30
31/// The main render worker loop.
32async fn render_worker<R: Runtime>(
33    app_handle: AppHandle<R>,
34    mut rx: mpsc::UnboundedReceiver<RenderWorkerTask>,
35) {
36    while let Some(task) = rx.recv().await {
37        match task {
38            RenderWorkerTask::Render { id, entry } => {
39                let report = async {
40                    let widget_dir = app_handle.widgets().dir().join(&id);
41                    let code = Bundler::new(widget_dir, entry)?.bundle().await?;
42                    Ok::<_, anyhow::Error>(code)
43                }
44                .await
45                .into();
46                let event = RenderEvent { id: &id, report };
47                if let Err(e) = event.emit_to(&app_handle, DeskulptWindow::Canvas) {
48                    error!("Failed to emit RenderEvent for widget {id}: {e:?}");
49                };
50            },
51        }
52    }
53}
54
55/// Handle for communicating with the render worker.
56pub struct RenderWorkerHandle(mpsc::UnboundedSender<RenderWorkerTask>);
57
58impl RenderWorkerHandle {
59    /// Create a new [`RenderWorkerHandle`] instance.
60    ///
61    /// This immediately spawns a dedicated render worker on Tauri's singleton
62    /// async runtime that listens for incoming [`RenderWorkerTask`]s and
63    /// processes them asynchronously in order.
64    pub fn new<R: Runtime>(app_handle: AppHandle<R>) -> Self {
65        let (tx, rx) = mpsc::unbounded_channel();
66        tauri::async_runtime::spawn(async move {
67            render_worker(app_handle, rx).await;
68        });
69        Self(tx)
70    }
71
72    /// Instruct the render worker to process a task.
73    ///
74    /// This does not block. The task is sent to the render worker for
75    /// asynchronous processing and does not wait for completion. An error is
76    /// returned if task submission fails, but not task processing fails.
77    pub fn process(&self, task: RenderWorkerTask) -> Result<()> {
78        Ok(self.0.send(task)?)
79    }
80}
