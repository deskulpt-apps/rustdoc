22/// Manager for Deskulpt widgets.
23pub struct WidgetsManager<R: Runtime> {
24    /// The Tauri app handle.
25    app_handle: AppHandle<R>,
26    /// The widgets directory.
27    dir: PathBuf,
28    /// The widget catalog.
29    catalog: RwLock<WidgetCatalog>,
30    /// The handle for the render worker.
31    render_worker: RenderWorkerHandle,
32    /// The setup state for frontend windows.
33    setup_state: SetupState,
34}
35
36impl<R: Runtime> WidgetsManager<R> {
37    /// Initialize the [`WidgetsManager`].
38    ///
39    /// The catalog is initialized as empty. When all windows complete setup,
40    /// an initial refresh would be triggered by [`Self::complete_setup`] to
41    /// populate the catalog. A render worker is started immediately.
42    pub fn new(app_handle: AppHandle<R>) -> Result<Self> {
43        let dir = if cfg!(debug_assertions) {
44            app_handle.path().resource_dir()?
45        } else {
46            app_handle.path().document_dir()?.join("Deskulpt")
47        };
48        let dir = dunce::simplified(&dir).join("widgets");
49        std::fs::create_dir_all(&dir)?;
50
51        let render_worker = RenderWorkerHandle::new(app_handle.clone());
52
53        Ok(Self {
54            app_handle,
55            dir,
56            catalog: Default::default(),
57            render_worker,
58            setup_state: Default::default(),
59        })
60    }
61
62    /// Get the widgets directory.
63    pub fn dir(&self) -> &Path {
64        &self.dir
65    }
66
67    /// Reload a specific widget by its ID.
68    ///
69    /// This method loads the widget manifest from the corresponding widget
70    /// directory and updates the catalog entry for that widget. This could be
71    /// an addition, removal, or modification. It then syncs the settings with
72    /// the updated catalog. If any step fails, an error is returned.
73    pub fn reload(&self, id: &str) -> Result<()> {
74        let widget_dir = self.dir.join(id);
75        let manifest = WidgetManifest::load(&widget_dir);
76
77        let mut catalog = self.catalog.write();
78        if let Some(manifest) = manifest.transpose() {
79            catalog.0.insert(id.to_string(), manifest.into());
80        } else {
81            catalog.0.remove(id);
82        }
83        UpdateEvent(&catalog).emit(&self.app_handle)?;
84
85        self.app_handle
86            .settings()
87            .update_with(|settings| catalog.compute_settings_patch(settings))?;
88        Ok(())
89    }
90
91    /// Reload all widgets.
92    ///
93    /// This method loads a new widget catalog from the widgets directory and
94    /// replaces the existing catalog. It then syncs the settings with the
95    /// updated catalog. If any step fails, an error is returned.
96    pub fn reload_all(&self) -> Result<()> {
97        let new_catalog = WidgetCatalog::load(&self.dir)?;
98
99        let mut catalog = self.catalog.write();
100        *catalog = new_catalog;
101        UpdateEvent(&catalog).emit(&self.app_handle)?;
102
103        self.app_handle
104            .settings()
105            .update_with(|settings| catalog.compute_settings_patch(settings))?;
106        Ok(())
107    }
108
109    /// Render a specific widget by its ID.
110    ///
111    /// This method submits a render task for the specified widget to the render
112    /// worker. If the widget does not exist in the catalog or if task
113    /// submission fails, an error is returned. This method is non-blocking and
114    /// does not wait for the task to complete.
115    pub fn render(&self, id: &str) -> Result<()> {
116        let catalog = self.catalog.read();
117        let config = catalog
118            .0
119            .get(id)
120            .ok_or_else(|| anyhow!("Widget {id} does not exist in the catalog"))?;
121
122        if let Outcome::Ok(config) = config {
123            self.render_worker.process(RenderWorkerTask::Render {
124                id: id.to_string(),
125                entry: config.entry.clone(),
126            })?;
127        }
128        Ok(())
129    }
130
131    /// Render all widgets in the catalog.
132    ///
133    /// This method submits render tasks for all widgets in the catalog to the
134    /// render worker. If any task submission fails, an error containing all
135    /// accumulated errors is returned. This method is non-blocking and does not
136    /// wait for the tasks to complete.
137    pub fn render_all(&self) -> Result<()> {
138        let catalog = self.catalog.read();
139
140        let mut errors = vec![];
141        for (id, config) in catalog.0.iter() {
142            if let Outcome::Ok(config) = config
143                && let Err(e) = self.render_worker.process(RenderWorkerTask::Render {
144                    id: id.clone(),
145                    entry: config.entry.clone(),
146                })
147            {
148                errors.push(e.context(format!("Failed to send render task for widget {id}")));
149            }
150        }
151
152        if !errors.is_empty() {
153            let message = errors
154                .into_iter()
155                .map(|e| format!("{e:?}"))
156                .collect::<Vec<_>>()
157                .join("\n");
158            bail!(message);
159        }
160
161        Ok(())
162    }
163
164    /// Refresh a specific widget by its ID.
165    ///
166    /// This is equivalent to reloading that widget with [`Self::reload`] then
167    /// rendering it with [`Self::render`].
168    ///
169    /// Tauri command: [`crate::commands::refresh`].
170    pub fn refresh(&self, id: &str) -> Result<()> {
171        self.reload(id)?;
172        self.render(id)?;
173        Ok(())
174    }
175
176    /// Refresh all widgets.
177    ///
178    /// This is equivalent to reloading all widgets with [`Self::reload_all`]
179    /// then rendering all widgets with [`Self::render_all`].
180    ///
181    /// Tauri command: [`crate::commands::refresh_all`].
182    pub fn refresh_all(&self) -> Result<()> {
183        self.reload_all()?;
184        self.render_all()?;
185        Ok(())
186    }
187
188    /// Mark a window as having completed setup.
189    ///
190    /// If all windows have completed setup after this call, an initial refresh
191    /// of all widgets is trigger via [`Self::refresh_all`].
192    ///
193    /// Tauri command: [`crate::commands::complete_setup`].
194    pub fn complete_setup(&self, window: WebviewWindow<R>) -> Result<()> {
195        let window = window.label().try_into().unwrap();
196        let complete = self.setup_state.complete(window);
197        if complete {
198            self.refresh_all()?;
199        }
200        Ok(())
201    }

203    /// Add starter widgets if not already added.
204    ///
205    /// If the starter widgets have not been marked as added, this method will
206    /// copy the starter widgets from the bundled resources to the widgets base
207    /// directory. Failure to add individual starter widgets will be logged as
208    /// errors, but will not prevent others from being added, and will not cause
209    /// this method to return an error. However, only if all starter widgets are
210    /// added successfully will the settings be updated to mark them as added.
211    ///
212    /// This method is idempotent. If all starter widgets have been successfully
213    /// added once, subsequent calls are no-ops. If some starter widgets have
214    /// been added but not all, subsequent calls will silently skip already
215    /// existing starter widgets and attempt to add the remaining ones.
216    pub fn maybe_add_starter(&self) -> Result<()> {
217        if self.app_handle.settings().read().starter_widgets_added {
218            return Ok(());
219        }
220
221        let resource_dir = self.app_handle.path().resource_dir()?;
222
223        let mut has_error = false;
224        for widget in ["welcome"] {
225            let widget_id = format!("@deskulpt-starter.{widget}");
226            let src = resource_dir.join(format!("resources/widgets/starter/{widget}"));
227            let dst = self.dir.join(&widget_id);
228            if dst.exists() {
229                debug!(%widget_id, "Starter widget already exists, skipping");
230                continue;
231            }
232
233            match copy_dir::copy_dir(&src, &dst)
234                .with_context(|| format!("Failed to add starter widget {widget_id}"))
235            {
236                Ok(_) => {
237                    info!(%widget_id, "Added starter widget");
238                },
239                Err(e) => {
240                    has_error = true;
241                    error!(
242                        error = ?e,
243                        %widget_id,
244                        src = %src.display(),
245                        dst = %dst.display(),
246                        "Failed to add starter widget",
247                    );
248                },
249            }
250        }
251
252        if !has_error {
253            self.app_handle.settings().update(SettingsPatch {
254                starter_widgets_added: Some(true),
255                ..Default::default()
256            })?;
257        }
258        Ok(())
259    }

261    /// Fetch the widgets registry index.
262    ///
263    /// Before fetching, this method ensures that the catalog is up-to-date by
264    /// reloading all widgets. This is necessary for the frontend to know which
265    /// widgets are already installed.
266    pub async fn fetch_registry_index(&self) -> Result<RegistryIndex> {
267        self.reload_all()?;
268
269        let cache_dir = self.app_handle.path().app_cache_dir()?;
270        let fetcher = RegistryIndexFetcher::new(&cache_dir);
271        fetcher.fetch().await
272    }

274    /// Preview a widget from the registry.
275    pub async fn preview(&self, widget: &RegistryWidgetReference) -> Result<RegistryWidgetPreview> {
276        RegistryWidgetFetcher::default().preview(widget).await
277    }
278
279    /// Install a widget from the registry.
280    ///
281    /// If the widget already exists locally, an error is returned. After
282    /// installation, the widget is automatically refreshed to update the
283    /// catalog and render it.
284    pub async fn install(&self, widget: &RegistryWidgetReference) -> Result<()> {
285        let id = widget.local_id();
286        let widget_dir = self.dir.join(&id);
287        if widget_dir.exists() {
288            bail!("Widget {id} already installed");
289        }
290
291        RegistryWidgetFetcher::default()
292            .install(&widget_dir, widget)
293            .await?;
294
295        self.refresh(&id)?;
296        Ok(())
297    }
298
299    /// Uninstall a widget from the registry.
300    ///
301    /// If the widget does not exist locally, an error is returned. After
302    /// uninstallation, the widget is automatically reloaded to remove it from
303    /// the catalog.
304    pub async fn uninstall(&self, widget: &RegistryWidgetReference) -> Result<()> {
305        let id = widget.local_id();
306        let widget_dir = self.dir.join(&id);
307        if !widget_dir.exists() {
308            bail!("Widget {id} is not installed");
309        }
310        tokio::fs::remove_dir_all(&widget_dir)
311            .await
312            .with_context(|| format!("Failed to remove directory {}", widget_dir.display()))?;
313
314        self.reload(&id)?;
315        Ok(())
316    }
317
318    /// Upgrade a widget from the registry.
319    ///
320    /// If the widget does not exist locally, an error is returned. After
321    /// upgrading, the widget is automatically refreshed to update the catalog
322    /// and render it.
323    pub async fn upgrade(&self, widget: &RegistryWidgetReference) -> Result<()> {
324        let id = widget.local_id();
325        let widget_dir = self.dir.join(&id);
326        if !widget_dir.exists() {
327            bail!("Widget {id} is not installed");
328        }
329
330        // TODO: We should ideally perform some form of backup to allow rollback
331        // on failure, to avoid leaving the widget in a broken state
332        tokio::fs::remove_dir_all(&widget_dir)
333            .await
334            .with_context(|| format!("Failed to remove directory {}", widget_dir.display()))?;
335
336        RegistryWidgetFetcher::default()
337            .install(&widget_dir, widget)
338            .await?;
339
340        self.refresh(&id)?;
341        Ok(())
342    }
