1//! Utilities for interacting with the widgets registry index.

3use std::path::{Path, PathBuf};
4
5use anyhow::{Context, Result, bail};
6use reqwest::header::{ETAG, IF_NONE_MATCH};
7use reqwest::{Client, Response, StatusCode};
8use serde::{Deserialize, Serialize};
9use tracing::{debug, instrument, warn};
10
11use crate::catalog::WidgetManifestAuthor;

13/// An entry for a specific release of a widget in the registry.
14#[derive(Debug, Serialize, Deserialize, specta::Type)]
15#[serde(rename_all = "camelCase")]
16struct RegistryEntryRelease {
17    /// The version string of the release.
18    version: String,
19    /// The publication datetime of the release, in ISO 8601 format.
20    published_at: String,
21    /// The SHA-256 digest of the release package.
22    ///
23    /// This is used to verify integrity but also an immutable identifier for
24    /// uniquely locating the released widget package.
25    digest: String,
26}
27
28/// An entry for a widget in the registry.
29#[derive(Debug, Serialize, Deserialize, specta::Type)]
30#[serde(rename_all = "camelCase")]
31struct RegistryEntry {
32    /// The publisher handle.
33    handle: String,
34    /// The widget ID.
35    ///
36    /// Note that this ID is unique only within the publisher's namespace.
37    id: String,
38    /// The name of the widget.
39    name: String,
40    /// The authors of the widget.
41    authors: Vec<WidgetManifestAuthor>,
42    /// A short description of the widget.
43    description: String,
44    /// The releases of the widget, ordered from newest to oldest.
45    releases: Vec<RegistryEntryRelease>,
46}
47
48/// The widgets registry index.
49#[derive(Debug, Serialize, Deserialize, specta::Type)]
50#[serde(rename_all = "camelCase")]
51pub struct RegistryIndex {
52    /// The API version.
53    api: i32,
54    /// The datetime when the index was generated, in ISO 8601 format.
55    generated_at: String,
56    /// The list of widgets in the registry.
57    widgets: Vec<RegistryEntry>,
58}
59
60/// A fetcher for the widgets registry index.
61pub struct RegistryIndexFetcher {
62    /// The HTTP client.
63    client: Client,
64    /// The cache directory.
65    cache_dir: PathBuf,
66    /// The path to the cached index file.
67    cache_path: PathBuf,
68    /// The path to the cached etag file.
69    etag_path: PathBuf,
70}
71
72impl RegistryIndexFetcher {
73    /// The static URL of the widgets registry index.
74    const URL: &str = "https://cdn.jsdelivr.net/gh/deskulpt-apps/widgets@registry/index.json";
75
76    /// Create a new [`RegistryIndexFetcher`] instance.
77    ///
78    /// This will automatically assign cache paths within the given cache
79    /// directory. A new HTTP client will be created to perform requests.
80    pub fn new(cache_dir: &Path) -> Self {
81        Self {
82            client: Client::new(),
83            cache_dir: cache_dir.to_path_buf(),
84            cache_path: cache_dir.join("widgets-registry-index.json"),
85            etag_path: cache_dir.join("widgets-registry-index.etag"),
86        }
87    }
88
89    /// Fetch the widgets registry index.
90    ///
91    /// This will use a cached etag to perform a conditional GET request. If the
92    /// registry index has not changed since the last fetch, the cached version
93    /// will be used if available and valid. Otherwise, a fresh copy will be
94    /// fetched and cached.
95    #[instrument(skip_all, level = "debug")]
96    pub async fn fetch(&self) -> Result<RegistryIndex> {
97        tokio::fs::create_dir_all(&self.cache_dir)
98            .await
99            .context("Failed to create cache directory")?;
100
101        let cached_etag = self.read_etag().await.unwrap_or_else(|e| {
102            warn!(
103                error = ?e,
104                path = %self.etag_path.display(),
105                "Failed to read cached etag; proceeding without it",
106            );
107            None
108        });
109
110        let mut request = self.client.get(Self::URL);
111        if let Some(etag) = cached_etag {
112            debug!(%etag, "Using cached etag");
113            request = request.header(IF_NONE_MATCH, etag);
114        }
115
116        let response = request
117            .send()
118            .await
119            .context("Failed to send HTTP request")?;
120
121        match response.status() {
122            StatusCode::OK => self.handle_ok(response).await,
123            StatusCode::NOT_MODIFIED => self.handle_not_modified().await,
124            status => {
125                bail!("HTTP request failed with status code {status}");
126            },
127        }
128    }
129
130    /// Read the cached registry index from disk.
131    async fn read_cache(&self) -> Result<RegistryIndex> {
132        let cache = tokio::fs::read(&self.cache_path)
133            .await
134            .context("Failed to read cache")?;
135        let index = serde_json::from_slice(&cache).context("Failed to deserialize cache")?;
136        Ok(index)
137    }
138
139    /// Read the cached etag from disk.
140    ///
141    /// Specially, if the etag file does not exists, this returns `Ok(None)`
142    /// instead of an error.
143    async fn read_etag(&self) -> Result<Option<String>> {
144        match tokio::fs::read_to_string(&self.etag_path).await {
145            Ok(etag) => Ok(Some(etag.trim().to_string())),
146            Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(None),
147            Err(e) => Err(e.into()),
148        }
149    }
150
151    /// Handle a 200 OK response.
152    ///
153    /// This will read the response body, deserialize it, and cache both the
154    /// body and the etag (if present) to disk. Failure to cache will not be
155    /// treated as an error.
156    async fn handle_ok(&self, response: Response) -> Result<RegistryIndex> {
157        let etag = response
158            .headers()
159            .get(ETAG)
160            .and_then(|v| v.to_str().ok())
161            .map(|v| v.to_string());
162
163        let body = response
164            .bytes()
165            .await
166            .context("Failed to read response body")?;
167        let index = serde_json::from_slice(&body).context("Failed to deserialize response body")?;
168
169        match tokio::fs::write(&self.cache_path, &body).await {
170            Ok(_) => debug!(path = %self.cache_path.display(), "Cached registry index"),
171            Err(e) => warn!(
172                error = ?e,
173                path = %self.cache_path.display(),
174                "Failed to cache registry index",
175            ),
176        }
177
178        if let Some(etag) = etag {
179            match tokio::fs::write(&self.etag_path, &etag).await {
180                Ok(_) => debug!(path = %self.etag_path.display(), "Cached etag"),
181                Err(e) => warn!(
182                    error = ?e,
183                    path = %self.etag_path.display(),
184                    "Failed to cache etag",
185                ),
186            }
187        }
188
189        Ok(index)
190    }
191
192    /// Handle a 304 Not Modified response.
193    ///
194    /// This will attempt to read the cached index from disk. If that fails, it
195    /// will fall back to performing a fresh fetch.
196    async fn handle_not_modified(&self) -> Result<RegistryIndex> {
197        match self.read_cache().await {
198            Ok(index) => {
199                debug!("Widgets registry index not modified; using cache");
200                return Ok(index);
201            },
202            Err(e) => warn!(
203                error = ?e,
204                path = %self.cache_path.display(),
205                "Received 304 Not Modified but failed to read from cache; retrying fresh fetch",
206            ),
207        }
208
209        let response = self
210            .client
211            .get(Self::URL)
212            .send()
213            .await
214            .context("Failed to send HTTP request")?;
215
216        match response.status() {
217            StatusCode::OK => self.handle_ok(response).await,
218            status => bail!("Fetching failed with status code {status}"),
219        }
220    }
221}
