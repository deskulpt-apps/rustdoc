18/// A reference to a widget in the registry.
19///
20/// These information uniquely and immutably identify a widget package in the
21/// widgets registry.
22#[derive(Debug, Deserialize, specta::Type)]
23#[serde(rename_all = "camelCase")]
24pub struct RegistryWidgetReference {
25    /// The publisher handle.
26    handle: String,
27    /// The widget ID.
28    ///
29    /// Note that this ID is unique only within the publisher's namespace.
30    id: String,
31    /// The SHA-256 digest of the widget package.
32    digest: String,
33}
34
35impl RegistryWidgetReference {
36    /// Get the local ID of the widget.
37    ///
38    /// It is in the format `@handle.id` in order to be globally unique, valid
39    /// as a file name, and human-readable. The prefixing `@` is used to avoid
40    /// *accidental* name collisions with purely local widgets.
41    pub fn local_id(&self) -> String {
42        format!("@{}.{}", self.handle, self.id)
43    }
44}

46/// A descriptor for a widget in the registry.
47#[derive(Debug)]
48struct RegistryWidgetDescriptor {
49    /// The full OCI reference of the widget package.
50    reference: Reference,
51    /// The layer descriptor of the widget package.
52    ///
53    /// There should be only one layer in the package. This layer contains the
54    /// actual widget files, compressed as a gzipped tarball.
55    layer: OciDescriptor,
56    /// The annotations of the widget package containing widget metadata.
57    annotations: Option<BTreeMap<String, String>>,
58}
59
60/// Preview information about a widget in the registry.
61#[derive(Debug, Default, Serialize, specta::Type)]
62#[serde(rename_all = "camelCase")]
63pub struct RegistryWidgetPreview {
64    /// The local ID of the widget.
65    ///
66    /// See [`RegistryWidgetReference::local_id`] for details.
67    id: String,
68    /// The size of the widget package in bytes.
69    size: u64,
70    /// The URL of the widget package in the registry.
71    registry_url: String,
72    /// The creation datetime of the widget package, in ISO 8601 format.
73    #[serde(default, skip_serializing_if = "Option::is_none")]
74    #[specta(type = String)]
75    created: Option<String>,
76    /// The git repository URL of the widget source code.
77    #[serde(default, skip_serializing_if = "Option::is_none")]
78    #[specta(type = String)]
79    git: Option<String>,
80    /// More information as in the widget manifest.
81    #[serde(flatten)]
82    manifest: WidgetManifest,
83}

85/// A fetcher for widgets from the registry.
86///
87/// Use [`RegistryWidgetFetcher::default`] to create a new instance, which will
88/// create a new OCI client internally.
89#[derive(Default)]
90pub struct RegistryWidgetFetcher(Client);
91
92impl RegistryWidgetFetcher {
93    /// The base URL of the widgets registry in GHCR.
94    const REGISTRY_BASE: &str = "ghcr.io/deskulpt-apps/widgets";
95
96    /// The expected artifact type of the widget packages.
97    const EXPECTED_ARTIFACT_TYPE: &str = "application/vnd.deskulpt.widget.v1";
98
99    /// Fetch the descriptor of a widget from the registry.
100    ///
101    /// This does not download the actual widget files, only the metadata. It
102    /// verifies that the artifact type, number of layers, and media type of the
103    /// layer are as expected.
104    async fn fetch(&self, widget: &RegistryWidgetReference) -> Result<RegistryWidgetDescriptor> {
105        let reference: Reference = format!(
106            "{}/{}/{}@{}",
107            Self::REGISTRY_BASE,
108            widget.handle,
109            widget.id,
110            widget.digest
111        )
112        .parse()?;
113
114        let (manifest, _) = self
115            .0
116            .pull_image_manifest(&reference, &RegistryAuth::Anonymous)
117            .await?;
118
119        if manifest.artifact_type.as_deref() != Some(Self::EXPECTED_ARTIFACT_TYPE) {
120            bail!(
121                "Expected artifact type {}, got {:?}",
122                Self::EXPECTED_ARTIFACT_TYPE,
123                manifest.artifact_type
124            );
125        }
126
127        let num_layers = manifest.layers.len();
128        if num_layers != 1 {
129            bail!("Expected only one layer; got {num_layers}",);
130        }
131
132        // Safe to unwrap because we have checked that there is one element
133        let layer = manifest.layers.into_iter().next().unwrap();
134        if !layer.media_type.ends_with("tar+gzip") {
135            bail!("Expected gzip-compressed tar; got {}", layer.media_type);
136        }
137
138        Ok(RegistryWidgetDescriptor {
139            reference,
140            layer,
141            annotations: manifest.annotations,
142        })
143    }
144
145    /// Install a widget from the registry into the given directory.
146    pub async fn install(&self, dir: &Path, widget: &RegistryWidgetReference) -> Result<()> {
147        let RegistryWidgetDescriptor {
148            reference, layer, ..
149        } = self.fetch(widget).await?;
150
151        let sized_stream = self.0.pull_blob_stream(&reference, &layer).await?;
152        let reader = StreamReader::new(sized_stream.stream);
153
154        let buf = BufReader::new(reader);
155        let gz = GzipDecoder::new(buf);
156        let mut ar = Archive::new(gz);
157        ar.unpack(dir).await?;
158
159        Ok(())
160    }
161
162    /// Preview metadata about a widget in the registry.
163    ///
164    /// This does not download the actual widget files, but only fetches the
165    /// widget package metadata.
166    pub async fn preview(&self, widget: &RegistryWidgetReference) -> Result<RegistryWidgetPreview> {
167        let RegistryWidgetDescriptor {
168            reference,
169            layer,
170            annotations,
171        } = self.fetch(widget).await?;
172
173        let mut preview = RegistryWidgetPreview {
174            id: widget.local_id(),
175            size: layer.size as u64,
176            registry_url: format!("https://{reference}"),
177            ..Default::default()
178        };
179
180        if let Some(mut annotations) = annotations {
181            preview.created = annotations.remove("org.opencontainers.image.created");
182            preview.git = annotations
183                .remove("org.opencontainers.image.source")
184                .and_then(|source| source.split('@').next().map(|s| s.to_string()));
185
186            // Manifest fields
187            preview.manifest.name = annotations
188                .remove("org.opencontainers.image.title")
189                .unwrap_or_default();
190            preview.manifest.version = annotations.remove("org.opencontainers.image.version");
191            preview.manifest.authors = annotations
192                .remove("org.opencontainers.image.authors")
193                .and_then(|authors| serde_json::from_str(&authors).ok());
194            preview.manifest.license = annotations.remove("org.opencontainers.image.licenses");
195            preview.manifest.description =
196                annotations.remove("org.opencontainers.image.description");
197            preview.manifest.homepage = annotations.remove("org.opencontainers.image.url");
198        }
199
200        Ok(preview)
201    }
