1//! Setup state of frontend windows.
2//!
3//! TODO: Move this to a separate high-level crate that depends on this crate
4//! because it is not specific to widgets.
5
6use std::sync::atomic::{AtomicU8, Ordering};
7
8use bitflags::bitflags;
9use deskulpt_common::window::DeskulptWindow;
10
11bitflags! {
12  /// Flags representing the setup state of frontend windows.
13  struct SetupFlags: u8 {
14    const CANVAS = 1 << 0;
15    const MANAGER = 1 << 1;
16  }
17}
18
19impl From<DeskulptWindow> for SetupFlags {
20    fn from(window: DeskulptWindow) -> Self {
21        match window {
22            DeskulptWindow::Canvas => SetupFlags::CANVAS,
23            DeskulptWindow::Manager => SetupFlags::MANAGER,
24        }
25    }
26}
27
28/// Setup state of frontend windows.
29///
30/// This is essentially an atomic bitmask corresponding to [`SetupFlags`]. Each
31/// set bit means that the corresponding window has completed setup.
32#[derive(Default)]
33pub struct SetupState(AtomicU8);
34
35impl SetupState {
36    /// Mark a window as having completed setup.
37    ///
38    /// Returns `true` if all windows have completed setup after this call.
39    pub fn complete(&self, window: DeskulptWindow) -> bool {
40        let flag: SetupFlags = window.into();
41        // Set the corresponding bit and retrieve the previous state; we only
42        // need AcqRel not SeqCst because we don't care about total ordering
43        let prev = self.0.fetch_or(flag.bits(), Ordering::AcqRel);
44        let current = SetupFlags::from_bits_truncate(prev) | flag;
45        current.is_all()
46    }
47}
