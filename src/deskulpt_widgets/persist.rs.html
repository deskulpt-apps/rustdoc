25/// Persisted representation of the widget catalog.
26#[serde_as]
27#[derive(Debug, Default, Deserialize)]
28pub struct PersistedWidgetCatalog(
29    #[serde_as(deserialize_as = "MapSkipError<_, _>")] pub BTreeMap<String, PersistedWidget>,
30);
31
32impl PersistedWidgetCatalog {
33    /// Load the persisted widget catalog from disk.
34    ///
35    /// If the file does not exist, empty catalog is returned. All other errors
36    /// will be propagated.
37    pub fn load(path: &Path) -> Result<Self> {
38        if !path.exists() {
39            return Ok(Default::default());
40        }
41        let file = File::open(path)?;
42        let reader = BufReader::new(file);
43        let catalog = serde_json::from_reader(reader)?;
44        Ok(catalog)
45    }
46}

48/// A view of the widget catalog for persistence.
49///
50/// The serialization format will follow the representation of
51/// [`PersistedWidgetCatalog`].
52#[derive(Debug)]
53pub struct PersistedWidgetCatalogView<'a>(pub &'a WidgetCatalog);
54
55impl<'a> PersistedWidgetCatalogView<'a> {
56    /// Persist the widget catalog to disk.
57    pub fn persist(&self, path: &Path) -> Result<()> {
58        let file = File::create(path)?;
59        let writer = BufWriter::new(file);
60        serde_json::to_writer(writer, self)?;
61        Ok(())
62    }
63}

65impl<'a> Serialize for PersistedWidgetCatalogView<'a> {
66    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
67    where
68        S: serde::Serializer,
69    {
70        use serde::ser::SerializeMap;
71
72        #[derive(Serialize)]
73        #[serde(rename_all = "camelCase")]
74        struct PersistedWidgetView<'a> {
75            settings: &'a WidgetSettings,
76        }
77
78        let mut map = serializer.serialize_map(Some(self.0.0.len()))?;
79        for (k, v) in self.0.0.iter() {
80            map.serialize_entry(
81                k,
82                &PersistedWidgetView {
83                    settings: &v.settings,
84                },
85            )?;
86        }
87        map.end()
88    }
89}

91/// Debounce duration for persistence.
92const PERSIST_DEBOUNCE: Duration = Duration::from_millis(500);
93
94/// The worker for persisting widgets.
95struct PersistWorker<R: Runtime> {
96    /// The Tauri app handle.
97    app_handle: AppHandle<R>,
98    /// The receiver for incoming persist notifications.
99    rx: mpsc::UnboundedReceiver<()>,
100    /// Whether a persist task is pending.
101    pending: bool,
102    /// The debounce timer for persistence.
103    debounce: Pin<Box<Sleep>>,
104}
105
106impl<R: Runtime> PersistWorker<R> {
107    /// Create a new [`PersistWorker`] instance.
108    fn new(app_handle: AppHandle<R>, rx: mpsc::UnboundedReceiver<()>) -> Self {
109        Self {
110            app_handle,
111            rx,
112            pending: false,
113            debounce: Box::pin(tokio::time::sleep(PERSIST_DEBOUNCE)),
114        }
115    }
116
117    /// Run the worker event loop.
118    ///
119    /// This function will run indefinitely until the worker channel is closed.
120    async fn run(mut self) {
121        loop {
122            tokio::select! {
123                _ = &mut self.debounce, if self.pending => {
124                    self.on_deadline();
125                },
126                task = self.rx.recv() => match task {
127                    Some(_) => self.handle_task(),
128                    None => break,
129                },
130            }
131        }
132    }
133
134    /// Fire the persist operation when the debounce timer elapses.
135    fn on_deadline(&mut self) {
136        self.pending = false;
137        if let Err(e) = self.app_handle.widgets().persist() {
138            error!("Failed to persist widgets: {e:?}");
139        }
140    }
141
142    /// Handle an incoming persist task.
143    fn handle_task(&mut self) {
144        self.pending = true;
145        self.debounce
146            .as_mut()
147            .reset(Instant::now() + PERSIST_DEBOUNCE);
148    }
149}

151/// Handle for communicating with the persistence worker.
152pub struct PersistWorkerHandle(mpsc::UnboundedSender<()>);
153
154impl PersistWorkerHandle {
155    /// Create a new [`PersistWorkerHandle`] instance.
156    ///
157    /// This immediately spawns a dedicated worker on Tauri's singleton async
158    /// runtime that listens for incoming notifications and processes them with
159    /// debouncing.
160    pub fn new<R: Runtime>(app_handle: AppHandle<R>) -> Result<Self> {
161        let (tx, rx) = mpsc::unbounded_channel();
162        tauri::async_runtime::spawn(async move {
163            PersistWorker::new(app_handle, rx).run().await;
164        });
165        Ok(Self(tx))
166    }
167
168    /// Instruct the worker to persist the widget catalog.
169    ///
170    /// This does not block. The task is sent to the worker for asynchronous
171    /// processing and does not wait for completion. The worker will debounce
172    /// multiple notifications within a short time frame. An error is returned
173    /// only if task submission fails, but not if task processing fails.
174    pub fn notify(&self) -> Result<()> {
175        Ok(self.0.send(())?)
176    }
177}
